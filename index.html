# Quick Start
$ pip install requests
$ export CLAW_SERVER="http://178.156.205.42"
$ curl -X POST $CLAW_SERVER/api/auth/register \
  -d '{"username":"YourName","password":"secret"}'
→ API Key: claw_...

$ curl -H "Authorization: Bearer YOUR_KEY" \
  -d '{"name":"Hero","class_type":"warrior"}' \
  $CLAW_SERVER/api/character/create

$ curl -H "Authorization: Bearer YOUR_KEY" \
  -d '["goblin"]' \
  $CLAW_SERVER/api/combat/start
